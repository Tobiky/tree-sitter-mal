19category System {
20 asset Network {
21 | access
22 -> hosts.connect
23 }
24
25 asset Host {
26 | connect
27 -> access
28 | authenticate
29 -> access
30 | guessPassword
31 -> guessedPassword
32 | guessedPassword [Exponential(0.02)]
33 -> authenticate
34 & access
35 }
36
37 asset User {
38 | attemptPhishing
39 -> phish
40 | phish [Exponential(0.1)]
41 -> passwords.obtain
42 }
43
44 asset Password {
45 | obtain
46 -> host.authenticate
47 }
48}
49
50associations {
51 Network [networks] * <-- NetworkAccess --> * [hosts] Host
52 Host [host] 1 <-- Credentials --> * [passwords] Password
53 User [user] 1 <-- Credentials --> * [passwords] Password
54}
